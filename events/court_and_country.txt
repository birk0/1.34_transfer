country_event = {
    id = cac.3
    title = "none"
    desc = "none"
    picture = "none"
    hidden = yes
    is_triggered_only = yes

    immediate = {
        clr_country_flag = in_court_and_country
        clr_country_flag = cac_promising_representative
        clr_country_flag = cac_bold_accusations
    }

    option = {
        if = {
            limit = {
                has_any_estates = yes
                NOT = {
                    any_estate_has_privilege = {
                        value = 1
                    }
                }
            }
            country_event = {
                id = cac.4 # Resolved State Conflict
            }
        }
        else_if = {
            limit = {
                NOT = { has_any_estates = yes }
                check_variable = {
                    which = absolutism_desire 
                    value = 15
                }
            }
            country_event = {
                id = cac.5 # An Absolutist Court
            }
        }
        else_if = {
            limit = {
                NOT = { has_any_estates = yes }
                check_variable = {
                    which = absolutism_desire
                    value = 5
                }
            }
            country_event = {
                id = cac.6 # Republican Values
            }
        }
        else_if = {
            limit = {
                NOT = { has_any_estates = yes }
            }
            country_event = {
                id = cac.7 # Liberalist Society
            }
        }
        else = {
            country_event = {
                id = cac.8 # Influential Estates
            }
        }
    }
}

country_event = {
    id = cac.9
    title = "cac.9.t"
    desc = {
        trigger = { NOT = { has_any_estates = yes } }
        desc = "cac.9.desc_republic" # Factions
    }
    desc = {
        trigger = { government = tribal }
        desc = "cac.9.desc_tribal" # Tribes
    }
    desc = {
        trigger = { technology_group = indian }
        desc = "cac.9.desc_indian" # Castes
    }
    desc = {
        trigger = { religion_group = muslim }
        desc = "cac.9.desc_muslim" # Ulemma
    }
    desc = {
        trigger = { religion_group = eastern }
        desc = "cac.9.desc_eastern" # Palace officials
    }
    desc = {
        trigger = { religion_group = pagan }
        desc = "cac.9.desc_pagan" # Tradtionalists
    }
    desc = {
        trigger = {
            NOT = { religion_group = pagan }
            NOT = { religion_group = eastern }
            NOT = { religion_group = muslim }
            NOT = { technology_group = indian }
            NOT = { government = tribal }
        }
        desc = "cac.9.d" # Estates
    }
    picture = "ADVISOR_eventPicture"

    is_triggered_only = yes

    mean_time_to_happen = {
        days = 1
        modifier = {
            factor = 0.25
            has_country_flag = authority_questioned
        }
    }

    trigger = {
        NOT = { num_of_rebel_armies = 1 }
        employed_advisor = {
            category = ADM
        }
        any_estate_has_privilege = {
            value = 1
        }
        NOT = { has_country_modifier = "cac_9_modifier" }
    }

    immediate = {
        clr_country_flag = authority_questioned
        hidden_effect = {
            add_country_modifier = {
                name = "cac_9_modifier"
                duration = 730
                hidden = yes
            }
        }
    }

    option = { #Order his arrest.
        name = "cac.9.a"
        ai_chance = {
            factor = 100
            modifier = {
                factor = 0
                has_any_estates = yes
                NOT = {
                    check_variable = {
                        which = estate_revoke_var
                        value = 1
                    }
                }
            }
        }
        if = {
            limit = { has_any_estates = yes } 
            allow_privilege_revoke_effect = yes
        }
        else = {
            change_variable = {
                which = absolutism_desire
                value = 2
            }
        }
        remove_advisor_by_category = ADM
        random_owned_province = {
            limit = {
                island = no
                if = {
                    limit = { owner = { ai = yes } }
                    same_continent = capital
                }
            } 
            start_revolt = {
                size = 3
            }
        }
        add_years_of_income = -1
        add_adm_power = -150
    }
    option = { #Appease him
        name = "cac.9.b"
        if = {
            limit = { has_any_estates = yes } 
            disable_privilege_revoke_effect = yes
        }
        else = {
            change_variable = {
                which = absolutism_desire
                value = -2
            }
            add_adm_power = 75
            add_years_of_income = 0.45
        }
        add_adm_power = 50
        add_years_of_income = 0.3
        set_country_flag = authority_questioned
        add_estate_loyalty = {
            estate = all
            loyalty = 5
        }
        random_estate = {
            add_influence = 5
            duration = 1095
            desc = none
        }
    }
}
