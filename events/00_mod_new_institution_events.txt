#Challenging the Warrior Elites (was old_institution_events.8)
country_event = {
	id = new_institution_events.28
	title = "new_institution_events.28.t"
	desc = "new_institution_events.28.d"
	picture = HORDE_ON_HORSEBACK_eventPicture

	is_triggered_only = yes #on_bi_yearly_3
	goto = feudalism_province
	
	trigger = {
		fire_once_for_country = { event = nie28 }
		NOT = { has_institution = feudalism }
		any_owned_province = { feudalism = 25 }
	}

	mean_time_to_happen = { days = 1 }

	immediate = {
		hidden_effect = {
			set_event_flag = { event = nie28 }
			random_owned_province = {
				limit = { feudalism = 25 }
				save_event_target_as = feudalism_province
			}
		}
	}

	option = {
		name = "new_institution_events.28.a"
		event_target:feudalism_province = {
			area = {
				limit = { 
					owned_by = ROOT
					NOT = { feudalism = 100 } 
				}
				add_institution_embracement = {
					value = 100
					which = "feudalism"
				}
			}
			spawn_rebels = {
				size = 1
				type = noble_rebels
			}
		}
		if = {
			limit = { has_estate = estate_nomadic_tribes }
			add_estate_loyalty = {
				estate = estate_nomadic_tribes
				loyalty = -10
			}
		}
		else = {
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = -10
			}
		}
		ai_chance = {
			factor = 10
		}
	}
	option = {
		name = "new_institution_events.28.b"
		event_target:feudalism_province = {
			area = {
				limit = { owned_by = ROOT }
				add_institution_embracement = {
					value = -25
					which = "feudalism"
				}
			}
		}
	}
}
