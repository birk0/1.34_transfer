namespace = flavor_mug
normal_or_historical_nations = yes

# The Marathi Revolt
country_event = {
	id = flavor_mug.1
	title = "flavor_mug.EVTNAME1"
	desc = "flavor_mug.EVTDESC1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = MUG
		religion_group = muslim
		is_year = 1650
		NOT = { exists = MAR } # Marathas
		any_owned_province = {
			OR = {
					area = maidan_area
					area = desh_area
					culture = marathi
				}
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = "flavor_mug.EVTOPTA1" # Crush the rebels
		every_owned_province = {
			limit = {
				OR = {
					area = maidan_area
					area = desh_area
					culture = marathi
				}
				is_capital = no
			}
			add_core = MAR
			add_province_modifier = {
				name = add_unrest_10_modifier
				duration = 1825
			}
		}
	}
}

#Integration with Indian Culture
country_event = {
	id = new_mughal_events.2
	title = new_mughal_events.2.t
	desc = new_mughal_events.2.d
	picture = FORT_eventPicture

	fire_only_once = yes

	trigger = {
		tag = MUG
		num_of_owned_provinces_with = {
			value = 4
			religion = hinduism
		}
	}

	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.7
			has_idea_group = humanist_ideas
		}
	}

	option = {
		name = new_mughal_events.2.a
		ai_chance = {
			factor = 80
		}

		add_dip_power = 50
		add_country_modifier = {
			name = fee_mug_integration_hindu_culture
			duration = 36500
		}
	}

	option = {
		name = new_mughal_events.2.b

		add_country_modifier = {
			name = fee_mug_hindu_conversion_policy
			duration = 7300
		}
	}

	option = {
		name = new_mughal_events.2.c

		add_stability_or_adm_power = yes
	}
}
