hagia_sophia = {
	# province it starts in
	start = 151
	
	# date built in real life (so anything built during game time will be there if you start a game after that date)
	date = 537.01.01
	
	#time to build
	time = {
		months = 0
	}
	
	#how much to build one
	build_cost = 0
	
	#can we move it?
	can_be_moved = no
	
	#time to move the project one unit of distance, in days
	move_days_per_unit_distance = 10
	
	#tier that the project starts at when first placed in the game (be that at game start or when built by a country as the game progresses)
	starting_tier = 2
	
	#project type
	type = monument

	#can we build it?
	build_trigger = {
		OR = {		
			AND = {
				religion = orthodox
				owner_or_harmonized_religion = yes
			}
			AND = {
				religion = coptic
				owner_or_harmonized_religion = yes
			}			
			AND = {
				religion_group = muslim
				owner_or_harmonized_religion = yes
			}
			AND = {
				religion = catholic
				owner_or_harmonized_religion = yes
			}
		}			
	}

	#what to do when it's built
	on_built = {
		if = {
			limit = { 
				has_global_flag = hagia_sophia_now_mosque 
			}
			show_ambient_object = hagia_sophia_minarets
		}
		else = {
			show_ambient_object = hagia_sophia
		}
	}
	
	#what to do when it's destroyed
	on_destroyed = {
		hide_ambient_object = hagia_sophia_minarets
		hide_ambient_object = hagia_sophia
	}

	#can our country use it?
	can_use_modifiers_trigger = {
		OR = {		
			AND = {

			}
		}
