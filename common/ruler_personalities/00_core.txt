# Diplomatic Behaviour:
#wants_betray_allies = yes										> Adds a flat reason for declining a Call to Arms.
#heretic_ally_acceptance = 10									> Adds or subtracts reasons for Ai to ally heretics.
#heathen_ally_acceptance = 20									> Adds or subtracts reasons for AI to ally heathens.
#alliance_acceptance = 10										> Adds or subtracts reasons for AI to accept alliances.
#other_ai_help_us_multiplier = 2								> Modifier which makes other AI diplomatically bias to the country.
#royal_marriage_acceptance = 20									> Adds or subtracts reasons for AI to accept royal marriage.
#other_ai_peace_term_bonus = 10									> Bias between AI which lets them have more favourable peace deals.
#other_ai_war_chance_multiplier = 0.5							> Modifies the chance of other AI declaring war on them.
#peace_desire = -10												> Adds or subtracts reasons for the AI to peace out of a war.
#random_insults = yes											> Makes the AI send more insults.
#wants_disclose_attack = yes									> Makes the AI inform the target that they are preparing to declare war.
#send_warning_desire = 50										> Adds or subtracts reasons for AI to send a warning


# Administrative / Religious behavior:
#wants_to_appoint_cardinals_in_his_own_land = yes				> Makes the AI want to appoint cardinals in own land.
#razing = yes													> Makes the AI want to raze land/raid coasts and pillage capitals.
#lower_autonomy = yes											> Makes the AI want to lower autonomy.
#wants_ecumenical_council = yes									> Ai will invest money into becoming the curia controller.
#wants_excommunicate = yes										> Makes the AI want to excommunicate people more frequently.
#wants_explorers = yes											> Makes the AI hire more conquistadors and explore more.

# Economic Behavior:
#wants_sell_provinces = yes										> Makes the AI want to sell provinces.
#wants_loans = yes												> Makes the AI less concerned about loans.
#gift_chance = 65												> Adds or subtracts reasons for AI to send gifts.
#knowledge_sharing = yes										> Whether the AI wants to accept knowledge sharing.
#wants_condottieri = yes										> Whether the AI wants to accept condottieri.
#wants_avoid_debase = yes										> Makes the AI not want to debase currency.
#building_budget_multiplier = 1.8								> Modifies the budget AI has for constructing buildings.
#wants_avoid_loans = yes										> Makes the AI want to avoid taking loans.
#wants_debase = yes												> Makes the AI want to debase currency.

# Military behavior:
#fair_fights = yes												> Ai will prefer to pick on someone their own size.
#underestimate_ae = yes											> Makes the AI care less about aggressive expansion.
#wants_avoid_ugly_borders = yes									> Makes the AI care about bordergore (probably doesnt work).
#wants_threaten_war = yes										> Makes the AI more likely to use the threaten war action.
#easy_war_chance_multiplier = 0.5								> Modifies the chance of AI declaring an 'easy' war.
#war_chance_multiplier = 0.5									> Modifies the chance of AI declaring wars.
#enemy_strength_multiplier = 1.2								> Modifies how strong the AI views a potential war target.
#wants_heathen_land = yes										> Makes AI want heathen land more.
#wants_heretic_land = yes										> Makes AI want heretic land more.
#different_religion_war_multiplier = 2							> Modifies the chance of AI declaring war against other religions.
#wants_gold = yes												> Makes the AI want to conquer gold mines more.
#wants_colonies = yes											> Makes the AI want to conquer colonial regions more.
#wants_land = yes												> Makes the AI want to conquer land around them.
